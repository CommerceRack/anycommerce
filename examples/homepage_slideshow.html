This code requires cycle.js

http://jquery.malsup.com/cycle/

put this into the homepage template:

HINT: Change blank.gif for images/yourfilename.jpg

Put the following code into your init file:

//cycle used for slideshow app.rq.push(['script',0,app.vars.baseURL+'cycle.js']); //add slideshow to homepage. app.rq.push(['templateFunction','productTemplate','onCompletes',function(P) { var $target = $('#wideSlideshow'); if($target.children().length > 1) { $('#wideSlideshow').cycle({ fx:'fade', speed:'slow', timeout: 5000, pager:'#slideshowNav', slideExpr: 'li' }); } }]);

add this to your css file:

/*slideshow*/ #slideshowContainer {position:relative; z-index:10; height:300px; min-width:720px; max-width:900px; margin:0 auto; overflow:hidden; z-index:5; } /* setting height/width and overflow hidden eliminates popping (all banners showing then hiding) */ #slideshowContainer ul, .slideshow li {list-style:none; margin:0; padding:0; display:inline; float:left; } /* the slides */ #slideshowNav { z-index: 50; position: absolute; bottom: 10px; left: 10px; width:100%; text-align:left; } #slideshowNav a { margin: 0 8px; font-size:.01em; text-decoration: none; display:inline-block; padding: 6px 8px; border: 1px solid #c0b3ad; color:#ffffff; background: #ffffff; cursor:pointer; -webkit-border-radius: 8px; border-radius: 8px; } #slideshowNav a:hover, #slideshowNav a.activeSlide { color:#bfd255; background: #ea5051; } #slideshowNav a:focus { outline: none; }
